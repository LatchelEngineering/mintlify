def run_query():
    success= False
    while not success:
        try:
            # Authorize Google Sheets service account
            os.environ["GOOGLE_APPLICATION_CREDENTIALS"]='/home/jsalgado/data-analytics-warehouse-VM_Key.json'
            creds = ServiceAccountCredentials.from_json_keyfile_name(filename='data-analytics-warehouse-VM_Key.json')
            client = gspread.authorize(creds)

            # Connect to PostgreSQL database and execute SQL query
            con,cur = pg_connection()
            cur.execute("""(SQL_QUERY_CONTENT)""")
            df=pd.DataFrame(cur.fetchall(), columns=[...])

            # Process dataframe and download Google Sheet data
            console_Actions_Data_sh = client.open_by_key(Console_Actions_Data_Key).worksheet('CONSOLE ACTIONS RAW DATA')
            new_cell = 'D' + str(len(console_Actions_Data_sh.col_values(1)))
            last_date_str=console_Actions_Data_sh.row_values(new_cell)[0]

            # Filter dataframe based on last date in Google Sheet
            date_dt2 = datetime.strptime(last_date_str, '%m/%d/%Y %H:%M:%S')
            df['time_action_taken']=pd.to_datetime(df['time_action_taken'])
            df=df[(df['time_action_taken']>date_dt2)]

            # Process datetime columns and fill null values
            df['time_action_taken'] = df['time_action_taken'].dt.strftime('%m/%d/%Y %H:%M:%S')
            df['user_added'] = df['user_added'].dt.strftime('%m/%d/%Y %H:%M:%S')
            df.fillna("", inplace=True)

            # Append processed dataframe to Google Sheet
            console_Actions_Data_sh = client.open_by_key(Console_Actions_Data_Key).worksheet('CONSOLE ACTIONS RAW DATA')
            new_cell = 'A' + str(len(console_Actions_Data_sh.col_values(1))+1)
            console_Actions_Data_sh.update (new_cell,df.values.tolist())

            success=True

        except Exception as e:
          print(f"Error occurred: {e}")
          print("Retrying in 10 seconds...")
          time.sleep(10)

# Run the main function
run_query()
