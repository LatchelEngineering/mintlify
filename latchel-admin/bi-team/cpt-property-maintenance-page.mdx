### **Code Details**

1. `code.gs`
   Google App Script code responsible for interacting with Google Cloud's BigQuery to manage and retrieve data. Key functions include:
   1. `doGet()` - Entry point of a web app.
   2. `getPropertyManagers()`, `getPropertyIds(managerId)`, `getUnits(managerId, address)` - These functions send SQL-like queries to BigQuery, fetching distinct values from the 'data-analytics-warehouse' project, 'cpt\_team' dataset, 'weekly\_reports' table.
   3. `getFormData(unitId)` - Fetches and processes unit information from the 'unit\_information\_history' table.
   4. `formatDate(dateStr)` - Convert date `dateStr` to ISO format.
   5. `updateFormData(data)` - Inserts data into the 'unit\_information\_history' table.
2. `PropertyForm.html`
   1. This is an HTML script constructing a web form for property maintenance. This form interacts with the Google App Script functions defined in `code.gs`:
      1. Form Elements - Include drop-down menus for property manager, property, unit, plus ledger-able, has GL code, has lease, addendum, paid status, missed opportunities, others, has future GL code, future date GL code, month detect, has future lease, future lease date, need to be checked, note, updated at.
      2. Javascript Functions - Manage form dynamics, data population, and call relevant GAS functions.
      JavaScript functions include:
   2. `loadManagers()` - Populates Property Managers.
   3. `populateManagerSelect(managers) and loadProperties()` - Fetches and populates Property Information.
   4. `loadUnits()` - Fetches and populates Unit Information.
   5. `loadPropertyData()` - Fetch and populate property data retrieved from `getFormData(unitId)`.
   6. `clearFormFields()` - Clears the form.
   7. `submitForm()` - Collates form data and invokes `updateFormData(formData)` to update the record.

function getPropertyManagers() {

const datasetName = "cpt_team";
const tableName = "weekly_reports";

FROM \`${projectId}.${datasetName}.${tableName}\`
`;

query: query,
useLegacySql: false
};
const response = BigQuery.Jobs.query(request, projectId);


const rows = response.rows;
Logger.log(rows);
if (rows) {
const propertyManagers = rows.map(row => ({ name: row.f[0].v }));
return propertyManagers;
} else {
return [];
}
}


function getPropertyIds(managerId) {
  const projectId = "data-analytics-warehouse";
  const datasetName = "cpt_team";
  const tableName = "weekly_reports";

  const query = `
  SELECT distinct(property_address) as label
  FROM \`${projectId}.${datasetName}.${tableName}\` a
  WHERE cast(a.pm_company_id as string) = "${managerId}"
  `;

  const request = {
    query: query,
    useLegacySql: false
  };
  
  const response = BigQuery.Jobs.query(request, projectId);
  
  const rows = response.rows;
  Logger.log(rows);
  if (rows) {
    const properties = rows.map(row => ({ label: row.f[0].v }));
    return properties;
  } else {
    return [];
  }
}


function getUnits(managerId, address) {
  const projectId = "data-analytics-warehouse";
  const datasetName = "cpt_team";
  const tableName = "weekly_reports";

  const query = `
  SELECT distinct(property_id) as id, unit as label
  FROM \`${projectId}.${datasetName}.${tableName}\` a
  WHERE cast(a.pm_company_id as string) = "${managerId}" AND a.property_address = "${address}"
  `;
  
  const request = {
    query: query,
    useLegacySql: false
  };
  
  const response = BigQuery.Jobs.query(request, projectId);
  
  const rows = response.rows;
  Logger.log(rows);
  if (rows) {
    const units = rows.map(row => ({ id: row.f[0].v, label: row.f[1].v }));
    return units;
  } else {
    return [];
  }
}

function getFormData(unitId) {
    try {
        const projectId = "data-analytics-warehouse";
        const datasetName = "cpt_team";
        const tableName = "unit_information_history";

        const query = `
            SELECT
                property_id,
                ledgerable,
                has_gl_code,
                has_lease,
                addendum,
                has_paid,
                missed_opportunities,
                others,
                has_future_gl_code,
                future_date_gl_code,
                month_detect,
                has_future_lease,
                future_lease_date,
                need_to_be_checked,
                notes,
                updated_at,
                record_date
            FROM \`${projectId}.${datasetName}.${tableName}\`
            WHERE property_id = @unitId
        `;

        const queryParameters = [
            {
                name: 'unitId',
                parameterType: { type: 'STRING' },
                parameterValue: { value: unitId }
            }
        ];

        const request = {
            query: query,
            useLegacySql: false,
            queryParameters: queryParameters
        };
    
        const response = BigQuery.Jobs.query(request, projectId);
        const rows = response.rows;

        if (rows) {
            return {
                id: rows[0].f[0].v,
                ledgerable: rows[0].f[1].v,
                has_gl_code: rows[0].f[2].v,
                has_lease: rows[0].f[3].v,
                addendum: rows[0].f[4].v,
                has_paid: rows[0].f[5].v,
                missed_opportunities: rows[0].f[6].v,
                others: rows[0].f[7].v,
                has_future_gl_code: rows[0].f[8].v,
                future_date_gl_code: formatDate(rows[0].f[9].v),
                month_detect: formatDate(rows[0].f[10].v),

        throw error;
    }
}

function formatDate(dateStr) {
if (!dateStr) {
return null;
}
const dateObj = new Date(dateStr);
return dateObj.toISOString().substring(0, 10);
}

//populateManagerSelect(managers) and loadProperties() - Fetches and populates Property Information.
function populateManagerSelect(managers) {
const managerSelect = document.getElementById("managerId");
managerSelect.innerHTML =
'<option value="">Select a property manager</option>' +
managers.map(manager => `<option value="${manager.name}">${manager.name}</option>`).join('');
managerSelect.onchange = loadProperties;
}
function loadProperties() {
   const managerId = document.getElementById("managerId").value;
   if (managerId) {
     google.script.run.withSuccessHandler(populatePropertySelect).getPropertyIds(managerId);
     clearFormFields();
   }
 }
