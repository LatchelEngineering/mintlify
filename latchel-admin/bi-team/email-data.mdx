## Code

```text
#!/usr/bin/env python3

from oauth2client.service_account import ServiceAccountCredentials
import requests
import pandas as pd
import gspread
import os
import json
import numpy as np
from google.cloud import bigquery

# Load Google credentials
creds = ServiceAccountCredentials.from_json_keyfile_name(filename='data-analytics-warehouse-VM_Key.json')
client = gspread.authorize(creds)

# ZenDesk credentials and API endpoint
credentials = ('EMAIL_ADDRESS', 'API_KEY')
session = requests.Session()
session.auth = credentials
zendesk = 'https://latchel.zendesk.com/'

# Fetch data from the ZenDesk API and store it in a dataframe
df_metrics = pd.DataFrame()
next_page = 1
for page_iter in range(1000):
    url = f'{zendesk}/api/v2/ticket_metrics.json?page={next_page}'
    response = session.get(url)
    if response.status_code != 200:
        print(f'Error with status code {response.status_code}')
        exit()
    data = response.json()
    metrics_dict = data['ticket_metrics']
    df_metrics = pd.concat([df_metrics, pd.DataFrame(metrics_dict)])
    next_page = next_page + 1

# Process the 'df_metrics' dataframe
df_metrics.drop(columns=[...], inplace=True)

# Upload the processed dataframe to Google BigQuery table
os.environ["GOOGLE_APPLICATION_CREDENTIALS"] = '/home/jsalgado/data-analytics-warehouse-VM_Key.json'
client = bigquery.Client()
table_id = "data-analytics-warehouse.latchel_stage_bigquery.zendesk_tickets_metrics"
job_config = bigquery.LoadJobConfig(schema=[...])
job = client.load_table_from_dataframe(df_metrics, table_id, job_config=job_config)
job.result()

# Print the result of the operation
table = client.get_table(table_id)
print(
    "Loaded {} rows and {} columns to {}".format(
        table.num_rows, len(table.schema), table_id
    )
)
```
