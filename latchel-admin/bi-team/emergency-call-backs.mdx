df = df[(df['created_at']>date_dt2)]

# Sort dataframe by 'created_at' column
df = df.sort_values(by="created_at")

# Append the processed and filtered dataframe to the BigQuery table
table_id = "data-analytics-warehouse.latchelstage_bigquery.emergency_call_back"
job_config = bigquery.LoadJobConfig(schema=[...])
job = client.load_table_from_dataframe(df, table_id, job_config=job_config)
job.result()

# Print the result of the operation
table = client.get_table(table_id)
print(
    "Loaded {} rows and {} columns to {}".format(
        table.num_rows, len(table.schema), table_id
